def create_sequences(data, lookback):
    X, y = [], []
    for i in range(lookback, len(data)):
        X.append(data[i - lookback:i])
        y.append(data[i])
    return np.array(X), np.array(y)

In[8]:
def evaluate(y_true, y_pred):
    mae = mean_absolute_error(y_true, y_pred)
    mse = mean_squared_error(y_true, y_pred)
    rmse = np.sqrt(mse)
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100
    r2 = r2_score(y_true, y_pred)
    return mae, mse, rmse, mape, r2

In[9]:
def train_lstm(X_train, y_train, X_val, y_val, units=64, epochs=50):
    model = Sequential([
        LSTM(units, input_shape=(X_train.shape[1], X_train.shape[2])),
        Dense(1)
    ])
    model.compile(optimizer='adam', loss='mse')
    model.fit(X_train, y_train,
              validation_data=(X_val, y_val),
              epochs=epochs,
              batch_size=32,
              callbacks=[EarlyStopping(patience=5, restore_best_weights=True)],
              verbose=0)
    return model

for idx, (tr, vl, te) in enumerate(ratios):
    print(f"\n== T l {tr:.2f}-{vl:.2f}-{te:.2f} ==")


    scaler = MinMaxScaler()
    y_scaled = scaler.fit_transform(y_raw)


    n_total = len(y_scaled)
    n_train = int(tr * n_total)
    n_val = int(vl * n_total)
    n_test = n_total - n_train - n_val

    train = y_scaled[:n_train]
    val = y_scaled[n_train:n_train + n_val]
    test = y_scaled[n_train + n_val:]

    dates_train = df.index[:n_train]
    dates_val = df.index[n_train:n_train + n_val]
    dates_test = df.index[n_train + n_val:]


    X_train, y_train = create_sequences(train, lookback)
    X_val, y_val = create_sequences(np.concatenate([train[-lookback:], val]), lookback)
    X_test, y_test = create_sequences(np.concatenate([val[-lookback:], test]), lookback)

    model = train_lstm(X_train, y_train, X_val, y_val)


    y_train_pred = model.predict(X_train).squeeze()
    y_val_pred = model.predict(X_val).squeeze()
    y_test_pred = model.predict(X_test).squeeze()


    y_train_true = scaler.inverse_transform(y_train.reshape(-1, 1)).squeeze()
    y_val_true = scaler.inverse_transform(y_val.reshape(-1, 1)).squeeze()
    y_test_true = scaler.inverse_transform(y_test.reshape(-1, 1)).squeeze()

    y_train_pred = scaler.inverse_transform(y_train_pred.reshape(-1, 1)).squeeze()
    y_val_pred = scaler.inverse_transform(y_val_pred.reshape(-1, 1)).squeeze()
    y_test_pred = scaler.inverse_transform(y_test_pred.reshape(-1, 1)).squeeze()

    mae, mse, rmse, mape, r2 = evaluate(y_test_true, y_test_pred)
    print(f" Kt qu nh gi (Test Set):")
    print(f"   MAE :  {mae:.4f}")
    print(f"   MSE :  {mse:.4f}")
    print(f"   RMSE:  {rmse:.4f}")
    print(f"   MAPE: {mape:.2f}%")
    print(f"   R   : {r2:.4f}")

    if r2 > best_score:
        best_score = r2
        best_model = model
        best_scaler = scaler
        best_data = y_scaled
        best_split = (tr, vl, te)
        best_lookback = lookback

    history_results.append((f"{tr:.2f}-{vl:.2f}-{te:.2f}", r2))

    train_dates = dates_train[lookback:]
    val_dates = dates_val
    test_dates = dates_test

    plt.figure(figsize=(16, 6))
    plt.plot(train_dates, y_train_true, label='Train', color='blue')

    plt.plot(val_dates, y_val_true, label='Validation', color='green')

    plt.plot(test_dates, y_test_true, label='Test', color='red')
    plt.plot(test_dates, y_test_pred, '--', label='D on (Test)', color='orange')

    metrics_text = (f"MAE: {mae:.2f} | MSE: {mse:.2f} | RMSE: {rmse:.2f}\n"
                    f"MAPE: {mape:.2f}% | R: {r2:.4f}")
    plt.text(test_dates[int(len(test_dates)*0.05)], max(y_test_true), metrics_text,
             bbox=dict(facecolor='white', alpha=0.8), fontsize=10)

    plt.title(f'LSTM: D on gi VIX - T l {tr:.2f}-{vl:.2f}-{te:.2f}')
    plt.xlabel('Ngy')
    plt.ylabel('Gi VIX')
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.show()
