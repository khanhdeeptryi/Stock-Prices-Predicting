def run_arima_evaluation_train_val_test(series, train_ratio, test_ratio, val_ratio=0.0, order=(1, 1, 0)):
    assert round(train_ratio + test_ratio + val_ratio, 2) == 1.0, "Tổng tỉ lệ phải bằng 1.0"
    n = len(series)

    train_size = int(n * train_ratio)
    val_size = int(n * val_ratio)
    test_size = int(n * test_ratio)

    train_end = train_size
    val_end = train_end + val_size
    test_end = val_end + test_size

    train = series[:train_end]
    val = series[train_end:val_end]
    test = series[val_end:test_end]

    train_val = pd.concat([train, val])

    model = ARIMA(train_val, order=order)
    model_fit = model.fit()
    forecast = model_fit.forecast(steps=len(test))

    mae = mean_absolute_error(test, forecast)
    mape = np.mean(np.abs((test - forecast) / test)) * 100
    mse = mean_squared_error(test, forecast)
    rmse = np.sqrt(mse)
    r2 = r2_score(test, forecast)

    return {
        'Train %': train_ratio,
        'Val %': val_ratio,
        'Test %': test_ratio,
        'MAE': round(mae, 2),
        'MAPE': round(mape, 2),
        'RMSE': round(rmse, 2),
        'MSE': round(mse, 2),
        'R2': round(r2, 4)
    }

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.arima.model import ARIMA
from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score

# Hàm vẽ biểu đồ đầy đủ
def plot_full_arima_forecast(series, train_ratio, test_ratio, val_ratio=0.0, order=(1, 1, 0)):
    n = len(series)

    train_size = int(n * train_ratio)
    val_size = int(n * val_ratio)
    test_size = int(n * test_ratio)

    train_end = train_size
    val_end = train_end + val_size
    test_end = val_end + test_size

    train = series[:train_end]
    val = series[train_end:val_end]
    test = series[val_end:test_end]

    train_val = pd.concat([train, val])

    # Huấn luyện mô hình
    model = ARIMA(train_val, order=order)
    model_fit = model.fit()
    forecast = model_fit.forecast(steps=len(test))

    # Tính chỉ số đánh giá
    mae = mean_absolute_error(test, forecast)
    mape = np.mean(np.abs((test - forecast) / test)) * 100

    # Vẽ biểu đồ
    plt.figure(figsize=(14, 6))
    plt.plot(train.index, train, label='Train True', color='blue')
    plt.plot(val.index, val, label='Val True', color='green')
    plt.plot(test.index, test, label='Test True', color='red')
    plt.plot(test.index, forecast, label='Test Predicted', color='orange', linestyle='--')
    plt.title(f"ARIMA{order} - ({int(train_ratio*100)}/{int(val_ratio*100)}/{int(test_ratio*100)})")
    plt.xlabel("Index")
    plt.ylabel("Close Price")
    plt.legend()
    plt.grid(True, linestyle='--', alpha=0.6)
    plt.tight_layout()
    plt.show()
