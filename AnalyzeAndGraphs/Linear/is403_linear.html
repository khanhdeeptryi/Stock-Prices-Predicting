import pandas as pd
import numpy as np
from sklearn.decomposition import PCA
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import StandardScaler
from sklearn.pipeline import make_pipeline, Pipeline
from sklearn.metrics import mean_squared_error, r2_score, mean_absolute_error
from sklearn.compose import ColumnTransformer
import matplotlib.pyplot as plt

# 1. c d liu
df = pd.read_csv("/content/drive/MyDrive/IS403/DoAnCuoiKy/Dataset/VIX_Cleaned.csv")
df['Date'] = pd.to_datetime(df['Date'])
df = df.sort_values('Date')

# 2. Feature Engineering
df['Day_of_week'] = df['Date'].dt.dayofweek
df['Month'] = df['Date'].dt.month

# 3. X l Change nu  dng %
if df['Change'].max() < 1:
    df['Change'] *= 100

# 4. nh ngha bin
price_features = ['Open', 'High', 'Low']
other_features = ['Volume', 'Change', 'Day_of_week', 'Month']
all_features = price_features + other_features
target = 'Close'

X = df[all_features]
y = df[target]

# 5. To Preprocessing Pipeline
preprocessor = ColumnTransformer(transformers=[
    ('price_pca', make_pipeline(StandardScaler(), PCA(n_components=1)), price_features),
    ('other_scale', StandardScaler(), other_features)
])

# 6. Hm tnh metric
def calculate_metrics(y_true, y_pred):
    mse = mean_squared_error(y_true, y_pred)
    rmse = np.sqrt(mse)
    mae = mean_absolute_error(y_true, y_pred)
    mape = np.mean(np.abs((y_true - y_pred) / y_true)) * 100
    r2 = r2_score(y_true, y_pred)
    return mse, rmse, mae, mape, r2

def plot_prediction_chart(dates, y_train, y_test, y_val, y_pred_val, split_ratio):
    dates_train = dates[:len(y_train)]
    dates_test = dates[len(y_train):len(y_train) + len(y_test)]
    dates_val = dates[len(y_train) + len(y_test):]

    plt.figure(figsize=(14, 6))
    plt.plot(dates_train, y_train, label="Train", color='blue')
    plt.plot(dates_test, y_test, label="Test", color='orange')
    plt.plot(dates_val, y_val, label="Validate", color='green')
    plt.plot(dates_val, y_pred_val, label="Predictions_Val", color='purple', linestyle='--')

    plt.title(f"Linear Regression (PCA) - Ratio {split_ratio}")
    plt.xlabel("Date")
    plt.ylabel("Close Price")
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.show()


# 7. Hm chy m hnh vi cc t l khc nhau
def run_pipeline_with_split(train_ratio, val_ratio, test_ratio):
    total_size = len(df)
    train_size = int(total_size * train_ratio)
    val_size = int(total_size * val_ratio)
    test_size = total_size - train_size - val_size

    # Chia d liu theo thi gian
    X_train = df.iloc[:train_size][all_features]
    X_val = df.iloc[train_size:train_size + val_size][all_features]
    X_test = df.iloc[train_size + val_size:][all_features]

    y_train = y.iloc[:train_size]
    y_val = y.iloc[train_size:train_size + val_size]
    y_test = y.iloc[train_size + val_size:]

    # Pipeline
    full_pipeline = Pipeline(steps=[
        ('preprocessor', preprocessor),
        ('regressor', LinearRegression())
    ])
    full_pipeline.fit(X_train, y_train)

    # D on trn tp test
    y_pred_test = full_pipeline.predict(X_test)

    # nh gi trn tp test
    mse, rmse, mae, mape, r2 = calculate_metrics(y_test, y_pred_test)

    # V biu 
    split_str = f"{int(train_ratio*100)} - {int(val_ratio*100)} - {int(test_ratio*100)}"
    dates_train = df['Date'].iloc[:train_size]
    dates_val = df['Date'].iloc[train_size:train_size + val_size]
    dates_test = df['Date'].iloc[train_size + val_size:]

    plt.figure(figsize=(14, 6))
    plt.plot(dates_train, y_train, label="Train True", color='blue')
    plt.plot(dates_val, y_val, label="Validation True", color='green')
    plt.plot(dates_test, y_test, label="Test True", color='red')
    plt.plot(dates_test, y_pred_test, label="Test Predicted", color='orange', linestyle='--')
    plt.title(f"Linear Regression (PCA) - Split {split_str}")
    plt.xlabel("Date")
    plt.ylabel("Close Price")
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.show()

    return {
        "Train - Val - Test": split_str,
        "MAE": mae,
        "MSE": mse,
        "RMSE": rmse,
        "MAPE": mape,
        "R": r2
    }



# 8. Chy th vi 3 t l khc nhau
splits = [
    (0.75, 0.10, 0.15),
    (0.70, 0.10, 0.20),
    (0.65, 0.10, 0.25)
]

results = []
for split in splits:
    result = run_pipeline_with_split(*split)
    results.append(result)

# 9. Hin th kt qu
results_df = pd.DataFrame(results)

# Lm trn & nh dng bng
results_df['MAE'] = results_df['MAE'].round(2)
results_df['MSE'] = results_df['MSE'].round(2)
results_df['RMSE'] = results_df['RMSE'].round(2)
results_df['MAPE'] = results_df['MAPE'].map(lambda x: f"{x:.2f}%")
results_df['R'] = results_df['R'].round(2)

# In ra bng kt qu
print("\n=== BNG NH GI TRN VALIDATION SET ===")
print(results_df.to_markdown(index=False))  # hoc .to_string(index=False)
